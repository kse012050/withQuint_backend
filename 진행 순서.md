1. git init 으로 node 세팅
2. npm i dotenv express morgan multer mysql2 설치
3. npm 1 -D nodemon 설치
4. "start": "nodemon app.js", 입력
5. app.js 세팅 
~~~js 
const express = require('express');
const morgan = require('morgan');
const dotenv = require('dotenv');

const app = express();

dotenv.config();

app.set('port', process.env.PORT || 8001);

app.use(morgan('dev'));
app.use(express.json());
app.use(express.urlencoded({ extended: false }));

app.get('/user', (req, res, next)=>{
    console.log(req);
    console.log(res);
})

app.use((req, res, next)=>{
    const error = new Error(`${req.method} ${req.url} 라우터가 없습니다.`)
    error.status = 404;
    next(error);
})


app.listen(app.get('port'), () => {
    console.log(app.get('port'), '번 포트에서 대기 중');
})
~~~

7. mySQL user 테이블 생성
~~~mysql
create table `users` (
 `id` int(11) NOT NULL AUTO_INCREMENT,
 `userId` varchar(10) NOT NULL,
 `password` varchar(60) NOT NULL,
 `nickname` varchar(20) NOT NULL,
 `mobile` int(11) NOT NULL,
 `termsOfService` CHAR(1) NOT NULL CHECK (termsOfService = 'y'),
 `privacyPolicy` CHAR(1) NOT NULL CHECK (privacyPolicy = 'y'),
 `created` datetime NOT NULL,
 PRIMARY KEY (`id`)
);
~~~
